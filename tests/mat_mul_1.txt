var a = [], b = [], c = [], n = 100;
for (var i = 0; i < n; i += 1)
{
    a.push([]);
    b.push([]);
    c.push([]);
    
    for (var j = 0; j < n; j += 1)
    {
        a[i].push(rand());
        b[i].push(rand());
        c[i].push(0);
    }
}

var start = clock();
for (var i = 0; i < n; i += 1)
{
    for (var j = 0; j < n; j += 1)
    {
        for (var k = 0; k < n; k += 1)
        {
            c[i][k] += a[i][j] * b[j][k];
        }
    }
}
println(clock() - start);
